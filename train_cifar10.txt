device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

# SIMPLIFIED MAGDrop Implementation (FIXED)
class MAGDrop(nn.Module):
    def __init__(self, base_p=0.3, beta=0.9, tau=0.1):
        super().__init__()
        self.base_p = base_p
        self.beta = beta
        self.tau = tau
        # No momentum tracking to avoid batch size issues
        self.running_grad_norm = 1.0
        
    def forward(self, x):
        if not self.training:
            return x
            
        # Simplified version: use random noise as gradient proxy
        # In practice, you'd use proper gradient hooks
        batch_size = x.size(0)
        fake_grad = torch.randn_like(x)
        
        # Compute gradient norm per sample
        grad_norm = torch.norm(fake_grad.view(batch_size, -1), dim=1)
        
        # Update running average
        self.running_grad_norm = 0.99 * self.running_grad_norm + 0.01 * grad_norm.mean().item()
        
        # Simplified MAGDrop: higher dropout when gradients are large
        p = self.base_p * (grad_norm / (self.running_grad_norm + 1e-8))
        p = torch.sigmoid(p / self.tau)  # Smooth the dropout rate
        p = p.clamp(0.1, 0.6)  # Reasonable bounds
        
        # Apply dropout
        mask = torch.bernoulli(1 - p.view(-1, 1, 1, 1).expand_as(x))
        return x * mask / (1 - p.mean())

# AGR Implementation
class AGR(nn.Module):
    def __init__(self, lambda_reg=0.01):
        super().__init__()
        self.lambda_reg = lambda_reg
    
    def forward(self, x):
        return x
    
    def regularization_loss(self, model):
        reg_loss = 0
        for param in model.parameters():
            if param.grad is not None:
                reg_loss += torch.norm(param.grad) ** 2
        return self.lambda_reg * reg_loss

# Modified ResNet-18 with different regularization options
class RegularizedResNet18(nn.Module):
    def __init__(self, num_classes=10, reg_type='none', dropout_p=0.3):
        super().__init__()
        self.reg_type = reg_type
        self.backbone = resnet18(pretrained=False, num_classes=num_classes)
        
        # Modify for CIFAR-10 (smaller images)
        self.backbone.conv1 = nn.Conv2d(3, 64, kernel_size=3, stride=1, padding=1, bias=False)
        self.backbone.maxpool = nn.Identity()  # Remove initial maxpool
        
        # Add regularization components
        if reg_type == 'dropout':
            self.dropout = nn.Dropout(dropout_p)
        elif reg_type == 'magdrop':
            # Add MAGDrop after each major block
            self.magdrop1 = MAGDrop(base_p=0.3, beta=0.9, tau=0.1)
            self.magdrop2 = MAGDrop(base_p=0.3, beta=0.9, tau=0.1)
            self.magdrop3 = MAGDrop(base_p=0.3, beta=0.9, tau=0.1)
        elif reg_type == 'agr':
            self.agr = AGR(lambda_reg=0.01)
        
    def forward(self, x):
        x = self.backbone.conv1(x)
        x = self.backbone.bn1(x)
        x = self.backbone.relu(x)
        
        # Apply regularization after first conv
        if self.reg_type == 'dropout' and self.training:
            x = self.dropout(x)
        elif self.reg_type == 'magdrop' and self.training:
            x = self.magdrop1(x)  # After first conv
        
        x = self.backbone.layer1(x)
        
        if self.reg_type == 'magdrop' and self.training:
            x = self.magdrop2(x)  # After layer1
            
        x = self.backbone.layer2(x)
        x = self.backbone.layer3(x)
        
        if self.reg_type == 'magdrop' and self.training:
            x = self.magdrop3(x)  # After layer3
            
        x = self.backbone.layer4(x)
        
        x = self.backbone.avgpool(x)
        x = torch.flatten(x, 1)
        x = self.backbone.fc(x)
        
        return x

# Training function
def train_model(model, train_loader, test_loader, epochs, reg_type, device):
    optimizer = optim.AdamW(model.parameters(), lr=0.001, weight_decay=1e-4)
    scheduler = CosineAnnealingLR(optimizer, T_max=epochs)
    criterion = nn.CrossEntropyLoss()
    
    best_test_acc = 0
    history = {'train_acc': [], 'test_acc': [], 'gen_gap': []}
    
    for epoch in range(epochs):
        # Training
        model.train()
        train_correct = 0
        train_total = 0
        
        for batch_idx, (data, target) in enumerate(train_loader):
            data, target = data.to(device), target.to(device)
            
            optimizer.zero_grad()
            output = model(data)
            loss = criterion(output, target)
            
            # Add AGR regularization if applicable
            if reg_type == 'agr':
                reg_loss = model.agr.regularization_loss(model)
                loss += reg_loss
            
            loss.backward()
            optimizer.step()
            
            # Track accuracy
            pred = output.argmax(dim=1)
            train_correct += (pred == target).sum().item()
            train_total += target.size(0)
        
        train_acc = 100 * train_correct / train_total
        
        # Testing
        model.eval()
        test_correct = 0
        test_total = 0
        
        with torch.no_grad():
            for data, target in test_loader:
                data, target = data.to(device), target.to(device)
                output = model(data)
                pred = output.argmax(dim=1)
                test_correct += (pred == target).sum().item()
                test_total += target.size(0)
        
        test_acc = 100 * test_correct / test_total
        gen_gap = train_acc - test_acc
        
        # Update history
        history['train_acc'].append(train_acc)
        history['test_acc'].append(test_acc)
        history['gen_gap'].append(gen_gap)
        
        # Update scheduler
        scheduler.step()
        
        # Print progress
        if (epoch + 1) % 10 == 0 or epoch == 0:
            print(f'Epoch {epoch+1:3d}/{epochs}: '
                  f'Train Acc: {train_acc:6.2f}%, '
                  f'Test Acc: {test_acc:6.2f}%, '
                  f'Gap: {gen_gap:5.2f}%')
        
        # Save best model
        if test_acc > best_test_acc:
            best_test_acc = test_acc
    
    return history, best_test_acc

# Main experiment function
def run_cifar10_experiments(epochs=50):
    print("=== CIFAR-10 Regularization Comparison ===")
    print("Methods: None, Dropout, AGR, MAGDrop")
    print(f"Epochs: {epochs}, Device: {device}")
    print("=" * 60)
    
    # Data transforms
    transform_train = transforms.Compose([
        transforms.RandomCrop(32, padding=4),
        transforms.RandomHorizontalFlip(),
        transforms.ToTensor(),
        transforms.Normalize((0.4914, 0.4822, 0.4465), (0.2023, 0.1994, 0.2010))
    ])
    
    transform_test = transforms.Compose([
        transforms.ToTensor(),
        transforms.Normalize((0.4914, 0.4822, 0.4465), (0.2023, 0.1994, 0.2010))
    ])
    
    # Load datasets
    train_set = torchvision.datasets.CIFAR10(
        root='./data', train=True, download=True, transform=transform_train)
    test_set = torchvision.datasets.CIFAR10(
        root='./data', train=False, download=True, transform=transform_test)
    
    train_loader = DataLoader(train_set, batch_size=128, shuffle=True, num_workers=2)
    test_loader = DataLoader(test_set, batch_size=256, shuffle=False, num_workers=2)
    
    print(f"Dataset loaded: {len(train_set)} train, {len(test_set)} test samples")
    
    # Test all regularization methods
    reg_types = ['none', 'dropout', 'agr', 'magdrop']
    results = {}
    
    for reg_type in reg_types:
        print(f"\n{'='*50}")
        print(f"Training with {reg_type.upper()} regularization")
        print(f"{'='*50}")
        
        # Create model
        model = RegularizedResNet18(num_classes=10, reg_type=reg_type).to(device)
        
        # Train and evaluate
        start_time = time.time()
        history, best_acc = train_model(model, train_loader, test_loader, epochs, reg_type, device)
        training_time = time.time() - start_time
        
        # Store results
        results[reg_type] = {
            'final_train_acc': history['train_acc'][-1],
            'final_test_acc': history['test_acc'][-1],
            'final_gen_gap': history['gen_gap'][-1],
            'best_test_acc': best_acc,
            'training_time': training_time
        }
        
        print(f"{reg_type.upper()} completed in {training_time:.1f}s")
        print(f"Best Test Accuracy: {best_acc:.2f}%")
    
    # Print final results
    print(f"\n{'='*60}")
    print("FINAL RESULTS SUMMARY")
    print(f"{'='*60}")
    print(f"{'Method':<12} {'Train Acc':<10} {'Test Acc':<10} {'Gen Gap':<10} {'Best Test':<10}")
    print(f"{'-'*60}")
    
    for reg_type in reg_types:
        res = results[reg_type]
        print(f"{reg_type.upper():<12} {res['final_train_acc']:<10.2f} {res['final_test_acc']:<10.2f} "
              f"{res['final_gen_gap']:<10.2f} {res['best_test_acc']:<10.2f}")
    
    return results

# Save results to CSV
def save_results_to_csv(results, filename='cifar10_results.csv'):
    with open(filename, 'w', newline='') as f:
        writer = csv.writer(f)
        writer.writerow(['Method', 'Final Train Acc', 'Final Test Acc', 'Gen Gap', 'Best Test Acc', 'Training Time'])
        for reg_type, res in results.items():
            writer.writerow([
                reg_type.upper(),
                f"{res['final_train_acc']:.2f}%",
                f"{res['final_test_acc']:.2f}%",
                f"{res['final_gen_gap']:.2f}%",
                f"{res['best_test_acc']:.2f}%",
                f"{res['training_time']:.1f}s"
            ])
    print(f"\nResults saved to {filename}")
